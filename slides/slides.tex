\section{Physical Models}


\begin{frame}{Finite Elements model}

\begin{center}
$M \ddot{q}(t) + C \dot{q}(t) + K q(t) = f(t)$
\end{center}

\end{frame}


\begin{frame}{Continous-time state-space model}

\begin{center}
$\left\{
\begin{array}{ll}
\dot{x}(t) = A_cx(t) + B_cf(t) \\
y(t) = C_cx(t) + D_cf(t)
\end{array}
\right.$
\end{center}

$x =
\begin{pmatrix}
\dot{q} \\
q
\end{pmatrix}$

\end{frame}


\begin{frame}{Discret-time state-space model}

\begin{center}
$\left\{
\begin{array}{ll}
x_{k+1} & = A_dx_k + B_df_k \\
y_k & = C_dx_k + B_df_k
\end{array}
\right.$

\vspace{5mm}

where

\vspace{5mm}

$\begin{array}{ll}
A_d = exp(A_c \Delta t) \\
B_d= (A_d - I) A^{-1}_c B_c \\
C_d = C_c \\
D_d = D_c
\end{array}$

\end{center}


\end{frame}
